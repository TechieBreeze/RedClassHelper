       ```dart
       /// Matches Chinese/English question numbering at line start:
       /// "1.", "1、", "1)", "（1）", "1）", "①", etc.
       final _questionBreakRE = RegExp(r'^\s*(?:\d{1,4}[.、）\)．]\s*|（\d{1,4}）|[①②③④⑤⑥⑦⑧⑨⑩])', multiLine: true);

       List<String> splitIntoQuestionBlocks(String rawText) {
         if (rawText.trim().isEmpty) return [];
         final lines = rawText.split('\n');
         final blocks = <List<String>>[];
         var current = <String>[];

         for (final line in lines) {
           if (_questionBreakRE.hasMatch(line.trimLeft()) && current.isNotEmpty) {
             blocks.add(List<String>.from(current));
             current = [line];
           } else {
             current.add(line);
           }
         }
         if (current.isNotEmpty) blocks.add(current);

         return blocks.map((b) => b.join('\n').trim()).where((b) => b.isNotEmpty).toList();
       }
       ```

          ```dart
          /// LLM 解析分支：分块 → 逐题 LLM 调用 → 自动确认
          Future<void> llmParse() async {
            if (state.extractedText.isEmpty) return;

            state = state.copyWith(
              phase: ImportPhase.llmParsing,
              progress: 0.0,
              clearError: true,
            );

            final blocks = splitIntoQuestionBlocks(state.extractedText);
            if (blocks.isEmpty) {
              state = state.copyWith(
                phase: ImportPhase.idle,
                error: '未能将文本拆分为题目块',
              );
              return;
            }

            final candidates = <ParseCandidate>[];
            final sources = <int, ParseSource>{};
            final db = await ref.read(appDatabaseProvider.future);
            var fallbackCount = 0;

            for (var i = 0; i < blocks.length; i++) {
              state = state.copyWith(
                progress: (i / blocks.length) * 0.95, // 0-95% during processing
              );

              ParseCandidate? candidate;
              ParseSource source;

              try {
                // Try LLM parse with built-in retry (3 attempts per HttpLlmClient)
                final llmResult = await ref.read(llmClientProvider).parse(
                  blocks[i],
                  bankName: state.bankName,
                );
                // Canonicalize answer (per RESEARCH.md Pitfall 1)
                final canonicalAnswer = canonicalizeAnswer(llmResult.answer).join();
                candidate = llmResult.copyWith(
                  answer: canonicalAnswer,
                  metadata: {
                    ...llmResult.metadata,
                    'source': 'llm',
                    'chunkIndex': i.toString(),
                  },
                  confidence: 0.9, // LLM results get high confidence
                );
                source = ParseSource.llm;
              } on LlmRetryExhaustedException catch (e) {
                // D-09: 3 retries exhausted → heuristic fallback for this specific chunk
                source = ParseSource.fallback;
                fallbackCount++;
                candidate = _fallbackParseSingle(blocks[i], i);

                // Write to parse_log (D-09)
                await _logParseEvent(
                  db: db,
                  jobId: state.jobId,
                  level: 'warn',
                  message: '第 ${i + 1} 题 LLM 失败，切换启发式兜底',
                  context: {'attempts': e.attempts, 'lastError': e.lastError, 'chunkIndex': i},
                );
              } on Exception catch (e) {
                // Unexpected error → fallback
                source = ParseSource.fallback;
                fallbackCount++;
                candidate = _fallbackParseSingle(blocks[i], i);

                await _logParseEvent(
                  db: db,
                  jobId: state.jobId,
                  level: 'error',
                  message: '第 ${i + 1} 题 LLM 异常: ${e.toString()}',
                  context: {'error': e.toString(), 'chunkIndex': i},
                );
              }

              if (candidate != null) {
                candidates.add(candidate);
                sources[candidates.length - 1] = source;
              }
            }

            if (candidates.isEmpty) {
              // All chunks failed — log and fall back to full heuristic
          ```
